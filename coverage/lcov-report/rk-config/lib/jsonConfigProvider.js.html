"use strict";
 
const path = require('path');
const Util = require('rk-utils');
 
class JsonConfigProvider {
    /**
     * JSON file config data source
     * @constructs JsonConfigProvider
     * @param {string} configDir - The base directory of config files
     * @param {string} baseName - The basename of the config file
     * @param {string} [envFlag] - Environment flag
     */
    constructor(configDir, baseName, envFlag) {
        this._configDir = configDir;
        this._baseName = baseName;
        this._envFlag = envFlag || process.env.NODE_ENV || 'development';
 
        /**
         * The default config file naming functor
         * @type {Function}
         * @public
         **/
        this.defaultNaming = function (baseName) {
            return baseName + '.default.json';
        };
 
        /**
         * The environment-aware config file naming functor
         * @type {Function}
         * @public
         **/
        this.envSpecificNaming = function (baseName, envFlag) {
            return baseName + '.' + envFlag + '.json';
        };    
    }
 
    /**
     * Start loading the config files
     * @returns {Promise}
     */
    load() {
        let defaultCfgPath = path.join(this._configDir, this.defaultNaming(this._baseName));
        let envCfgPath = path.join(this._configDir, this.envSpecificNaming(this._baseName, this._envFlag));
        
        function* readFiles() {
            let defaultCfg = (yield Util.fs.pathExists(defaultCfgPath)) ? (yield Util.fs.readJson(defaultCfgPath)) : {};
            let envCfg = (yield Util.fs.pathExists(envCfgPath)) ? (yield Util.fs.readJson(envCfgPath)) : {};
            
            return Util._.defaultsDeep(envCfg, defaultCfg);
        }
        
        return Util.co(readFiles);
    }
}
 
module.exports = JsonConfigProvider;
